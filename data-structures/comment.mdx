## TypeScript Type

```typescript
interface Comment {
  id: string;
  accountId: string;
  postUrl: string;
  commentText: string;
  status: 'scheduled' | 'pending' | 'verifying' | 'completed' | 'failed';
  commentUrl?: string | null;
  error?: string | null;
  createdAt?: string;
  completedAt?: string | null;
}
```
